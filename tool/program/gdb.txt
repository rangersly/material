gdb c/c++ 代码调试工具
	使用 g++ -g 生成调试文件，不可以加 -O 选项

	set [argc]		# 传入参数
	l n				# 列出源码
	r				# 开始运行到第一个断点
	b n 			# 在第n行打断点
	info break		# 查看断点
	c				# 继续运行到下一个断点
	s				# 单步执行
	n				# 单步执行,不进入函数
	p				# 输出变量或表达式的值
	set var []		# 修改变量的值
	q				# 退出
	bt				# 查看函数调用栈

使用gdb调试core文件
	使用ulimit -c unlimited 使挂掉的程序产生core文件

gdb demo -p [pid]		# 调试运行中的程序
